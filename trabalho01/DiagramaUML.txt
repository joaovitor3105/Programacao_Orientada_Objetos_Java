        Comida
        Municao
        Partes_Mecanicas
        Remedios
    }

    enum Status {
        Ativo
        Doente
        Ferido
        Morto
    }

    enum Habilidade {
        Engenharia
        Medicina
        Combate
        Exploracao
        Culinaria
        Diplomacia
        Hackamento
        Furtividade
    }
